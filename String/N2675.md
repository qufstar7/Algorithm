		for (int i = 0; i < T; i++) {

			String[] str = br.readLine().split(" ");
			int R = Integer.parseInt(str[0]);
			String S = str[1];

			for (int j = 0; j < S.length(); j++) {
				for (int k = 0; k < R; k++) {
